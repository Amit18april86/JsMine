define(['dojo/_base/declare',
        'dojo/topic',
        'dojo/text!./Widget.html',
        'dojo/_base/lang',
        'esri/dijit/HomeButton',
        'esri/geometry/Extent',
        'dojo/_base/html',
        'dojo/dom-construct',
        'qsjs/BaseWidget',
        'esri/toolbars/navigation',
        'dijit/registry',
        'dojo/query',
        'dojo/on',
        "dijit/_WidgetsInTemplateMixin",
        "dijit/Toolbar",
        "dijit/TooltipDialog",
        "dijit/Tooltip",
        "dijit/form/Button",
        "dijit/popup",
        "dojo/dom",
        "dojo/dom-attr",
        "dojo/_base/array",
        "dojo/mouse",
        "dojo/domReady!"
    ],
    function (declare,Topic,template,lang,HomeButton,Extent,html,domConstruct, BaseWidget,Navigation,registry,
              dojoquery,on,_WidgetsInTemplateMixin,Toolbar,TooltipDialog,Tooltip,Button,popup,dom,domAttr,array,mouse) {
 
        var navToolbar,myTooltipDialog;
        var layerInfo = [];
        var clazz = declare([BaseWidget,_WidgetsInTemplateMixin], {
 
            baseClass: 'jimu-widget-demo',
            name: 'Navigation',
            templateString: template,
 
            startup: function() {
                this.inherited(arguments);
                this._bindEvent();
                this._updateToolTip();
                var slider = dojo.query('.dijitSliderImageHandle');
                this.own(on(slider, mouse.leave, lang.hitch(this, this._updateToolTipLabel)));
                this.own(on(slider, mouse.enter, lang.hitch(this, this._updateToolTipLabel)));
 
                //receive notification to change navigation mode.
                Topic.subscribe("navigation-change", this._changeType);
            },
 
            _bindEvent:function() {
                var self = this;
 
                navToolbar = new Navigation(this.map);
                on(navToolbar, "onExtentHistoryChange", this.extentHistoryChangeHandler);
 
                dojoquery('zoomnextIcon').on("click", lang.hitch(function (evt) {
                    navToolbar.zoomToNextExtent();
                }));
 
                dojoquery('zoomprevIcon').on("click", lang.hitch(function (evt) {
                    navToolbar.zoomToPrevExtent();
                }));
            },
 
            _changeType: function(type) {
                var self=this;
                console.log(type);
                switch(type) {
                    case "pan":
                        navToolbar.activate(Navigation.PAN);
                        console.log("navigation changed to pan");
                        break;
                    case "zoomin":
                        break;
                    case "zoomout":
                        break;
                    case "previous":
                        break;
                    case "next":
                        break;
                    case "activate":
                        navToolbar.activate();
                        break;
                    case "deactivate":
                        navToolbar.deactivate();
                        console.log("navigation deactivate");
                        break;
                }
            },
 
            _updateToolTip: function(){
                var centerSlider = dojo.query('.dijitSliderImageHandle');
                if(centerSlider){
                    this.sliderTooltip = new Tooltip({
                        id:"sliderToolTip",
                        connectId: [centerSlider],
                        position: ['below'],
                        label: this.map.getScale().toString().split(".")[0]
                    });
                    this.sliderTooltip.startup();
                }
 
            },
 
            _updateToolTipLabel: function () {
                var self = this;
                this.sliderTooltip.set("label", this.map.getScale().toString().split(".")[0]);
            },
 
            extentHistoryChangeHandler: function () {
                registry.byId("prevExtent").disabled = navToolbar.isFirstExtent();
                registry.byId("nextExtent").disabled = navToolbar.isLastExtent();
            },
 
            _showNextExtent: function () {
                navToolbar.zoomToNextExtent();
            },
 
            _showFullExtent: function () {
                //navToolbar.zoomToFullExtent();
                var self = this;
                var initExtent = new esri.geometry.Extent(self.appConfig.map.mapOptions.extent);
                this.map.setExtent(initExtent);
            },
 
            _showPreviousExtent: function () {
                navToolbar.zoomToPrevExtent();
            },
 
            _moveUp:function(){
                this.map.panUp();
            },
 
            _moveDown:function(){
                this.map.panDown();
            },
 
            _moveLeft:function(){
                this.map.panLeft();
            },
 
            _moveRight:function(){
                this.map.panRight();
            },
 
            _enablePanning: function(){
                navToolbar.activate(Navigation.PAN);
            },
 
            _enableRubberZoomIn: function(){
                Topic.publish("disableDependent");
                navToolbar.activate(Navigation.ZOOM_IN);
            },
 
            _enableRubberZoomOut: function(){
                Topic.publish("disableDependent");
                navToolbar.activate(Navigation.ZOOM_OUT);
            }
 
        });
 
        clazz.hasStyle = false;
        clazz.hasUIFile = false;
        clazz.hasLocale = false;
        clazz.hasConfig = false;
        return clazz;
    });
